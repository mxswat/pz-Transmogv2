module TransmogV2 {
	imports { Base }

    item TransmoggedItem
	{
		Weight	=	0.01,
		Type	=	Normal,
		DisplayName	= Transmogged Clothing,
		Icon	=	TransmogBagOne,
	}

    item TransmoggedBag
	{
		Weight	=	0.01,
		Type	=	Normal,
		DisplayName	= Transmogged Clothing,
		Icon	=	TransmogBagOne,
	}

    item HideItem
	{
		Weight	=	0.01,
		Type	=	Normal,
		DisplayName	= Hide Clothing,
		Icon	=	HideClothing,
	}

    recipe Get Cosmetic Clothing
  	{
        keep [GetTransmoggableClothing],

        Result:TransmoggedItem,
        OnCreate:GetTransmogClothing,
        RemoveResultItem:true,
        Time:15,
        AnimNode:Disassemble,	   	   
        Category:Transmog,
    }

    recipe Get Hide Clothing
  	{
        keep [GetTransmoggableClothing],

        Result:HideItem,
        OnCreate:GetHideClothing,
        RemoveResultItem:true,
        Time:15,
        AnimNode:Disassemble,	   	   
        Category:Transmog,
    }

    recipe Get Cosmetic Bag
  	{
        keep [GetTransmoggableClothing],

        Result:TransmoggedBag,
        OnCreate:GetTransmogClothing,
        RemoveResultItem:true,
        Time:15,
        AnimNode:Disassemble,	   	   
        Category:Transmog,
    }
}
